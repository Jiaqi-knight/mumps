# testing what we just built
set(MUMPS_LIBRARIES mumps_common ${ORDERING_LIBS})
set(MUMPS_INCLUDE_DIRS ${CMAKE_SOURCE_DIR}/include)

set(_sfilename input_simpletest_real)
set(_dfilename input_simpletest_real)
set(_cfilename input_simpletest_cmplx)
set(_zfilename input_simpletest_cmplx)

foreach(a ${arith})
  add_executable(${a}simpletest ${a}simpletest.F)
  target_link_libraries(${a}simpletest PRIVATE ${a}mumps ${MUMPS_LIBRARIES} ${NUMERIC_LIBS})
  target_include_directories(${a}simpletest PRIVATE ${MUMPS_INCLUDE_DIRS} ${NUMERIC_INCDIRS})
  if(UNIX)
    add_test(NAME Example_${a} COMMAND sh -c "$<TARGET_FILE:${a}simpletest> < ${CMAKE_CURRENT_SOURCE_DIR}/${_${a}filename}")
    set_tests_properties(Example_${a} PROPERTIES TIMEOUT 15 RUN_SERIAL true)
  endif()

  add_executable(${a}simpletest_save_restore ${a}simpletest_save_restore.F)
  target_link_libraries(${a}simpletest_save_restore PRIVATE ${a}mumps ${MUMPS_LIBRARIES} ${NUMERIC_LIBS})
  target_include_directories(${a}simpletest_save_restore PRIVATE ${MUMPS_INCLUDE_DIRS} ${NUMERIC_INCDIRS})
  if(UNIX)
    add_test(NAME SaveRestore_${a} COMMAND sh -c "$<TARGET_FILE:${a}simpletest_save_restore> < ${CMAKE_CURRENT_SOURCE_DIR}/${_${a}filename}")
    set_tests_properties(SaveRestore_${a} PROPERTIES TIMEOUT 15 RUN_SERIAL true)
endif()
endforeach()
